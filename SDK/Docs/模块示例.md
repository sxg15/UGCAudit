# 模块示例

本文档给出三个常见模块的写法参考：文本审核、图片审核、OCR。示例重点展示输入输出约定，实际业务逻辑可以替换成自己的模型或规则。

## 示例一：文本审核模块

适用场景：敏感词、广告词、违规文本检测。

`module.json`：

```json
{
  "id": "custom.example.text-safety",
  "name": "示例文本审核",
  "kind": "text_safety",
  "summary": "检查文本中是否包含风险词。",
  "modelLabel": "无需模型",
  "icon": "file-text",
  "modelConfigured": true,
  "launch": {
    "launchType": "python",
    "command": "main.py",
    "args": ["--input", "{inputJson}", "--output", "{outputJson}"],
    "notes": "读取文本集合并输出审核结论。"
  },
  "parameters": [
    {
      "key": "riskWords",
      "name": "风险词",
      "description": "按行填写需要命中的词。",
      "parameterType": "stringList",
      "defaultValue": ["广告", "联系方式"],
      "required": false,
      "options": []
    }
  ]
}
```

核心处理逻辑：

```python
texts = input_data.get("inputs", {}).get("texts", {}).get("items", [])
risk_words = input_data.get("params", {}).get("riskWords", [])

matches = []
for item in texts:
    text = str(item.get("text", ""))
    hit_words = [word for word in risk_words if word and word in text]
    if hit_words:
        matches.append({
            "name": item.get("name", ""),
            "path": item.get("path", ""),
            "hitWords": hit_words
        })

output = {
    "status": "completed",
    "verdict": "review" if matches else "pass",
    "message": f"命中 {len(matches)} 个文本。" if matches else "未发现风险文本。",
    "processedFiles": len(texts),
    "matchedFiles": len(matches),
    "matches": matches,
    "reportSection": f"### 文本审核\n\n处理 {len(texts)} 条文本，命中 {len(matches)} 条。"
}
```

推荐连接：

```text
待测项目中所有文本 --文本数据--> 示例文本审核
开始 -> 示例文本审核 -> 输出结果
```

## 示例二：图片审核模块

适用场景：图片分类、图片合规检测、图片风险打标。

`module.json`：

```json
{
  "id": "custom.example.image-safety",
  "name": "示例图片审核",
  "kind": "image_safety",
  "summary": "检查图片文件名中是否包含风险标记。",
  "modelLabel": "无需模型",
  "icon": "shield-alert",
  "modelConfigured": true,
  "launch": {
    "launchType": "python",
    "command": "main.py",
    "args": ["--input", "{inputJson}", "--output", "{outputJson}"],
    "notes": "读取图片集合并输出审核结论。"
  },
  "parameters": [
    {
      "key": "blockedNames",
      "name": "风险文件名片段",
      "description": "文件名包含这些片段时进入人工复审。",
      "parameterType": "stringList",
      "defaultValue": ["risk", "blocked"],
      "required": false,
      "options": []
    }
  ]
}
```

核心处理逻辑：

```python
images = input_data.get("inputs", {}).get("images", {}).get("items", [])
blocked_names = input_data.get("params", {}).get("blockedNames", [])

matches = []
for image in images:
    name = str(image.get("name", "")).lower()
    hit = [word for word in blocked_names if str(word).lower() in name]
    if hit:
        matches.append({
            "name": image.get("name", ""),
            "path": image.get("path", ""),
            "hitRules": hit
        })

output = {
    "status": "completed",
    "verdict": "review" if matches else "pass",
    "message": f"命中 {len(matches)} 张图片。" if matches else "未发现风险图片。",
    "processedFiles": len(images),
    "matchedFiles": len(matches),
    "matches": matches,
    "reportSection": f"### 图片审核\n\n处理 {len(images)} 张图片，命中 {len(matches)} 张。"
}
```

推荐连接：

```text
待测项目中所有图片 --图片数据--> 示例图片审核
开始 -> 示例图片审核 -> 输出结果
```

## 示例三：输出图片集合的模块

适用场景：图片预处理、图片切片、图片标注、关卡预览图生成。模块生成新图片后，把这些图片继续传给下游图片审核或 OCR 模块。

`module.json`：

```json
{
  "id": "custom.example.image-producer",
  "name": "示例图片生成",
  "kind": "image_safety",
  "summary": "读取图片集合，生成处理后的图片集合。",
  "modelLabel": "无需模型",
  "icon": "folder-open",
  "modelConfigured": true,
  "dataOutputs": [
    {
      "handle": "images",
      "name": "处理后图片",
      "dataType": "imageCollection"
    }
  ],
  "launch": {
    "launchType": "python",
    "command": "main.py",
    "args": ["--input", "{inputJson}", "--output", "{outputJson}"],
    "notes": "生成新图片，并把图片集合交给下游模块。"
  },
  "parameters": []
}
```

核心处理逻辑：

```python
from pathlib import Path
import shutil

images = input_data.get("inputs", {}).get("images", {}).get("items", [])
artifact_dir = Path(input_data["stepArtifactDir"])
artifact_dir.mkdir(parents=True, exist_ok=True)

output_items = []
for image in images:
    source = Path(image["path"])
    target = artifact_dir / f"processed_{source.name}"
    shutil.copyfile(source, target)
    output_items.append({
        "path": str(target),
        "name": target.name,
        "extension": target.suffix.lstrip("."),
        "sourceAssetId": "custom.example.image-producer",
        "sourceAssetName": "示例图片生成",
        "relativePath": target.name
    })

output = {
    "status": "completed",
    "verdict": "pass",
    "message": f"生成 {len(output_items)} 张处理后图片。",
    "processedFiles": len(images),
    "matchedFiles": 0,
    "artifactCount": len(output_items),
    "outputs": {
        "images": {
            "dataType": "imageCollection",
            "items": output_items
        }
    },
    "reportSection": f"### 图片生成\n\n已生成 {len(output_items)} 张处理后图片。"
}
```

推荐连接：

```text
待测项目中所有图片 --图片数据--> 示例图片生成
示例图片生成 --处理后图片--> 示例图片审核
开始 -> 示例图片生成 -> 示例图片审核 -> 输出结果
```

关键点：

- `dataOutputs` 负责让流程图显示图片输出口。
- `outputs.images` 负责把实际图片集合交给下游模块。
- `outputs.images.items[].path` 要写生成后图片的真实路径。

## 示例四：OCR 模块

适用场景：从图片中提取文字，并把文字交给后续文本审核模块。

`module.json`：

```json
{
  "id": "custom.example.ocr",
  "name": "示例 OCR",
  "kind": "image_ocr",
  "summary": "从图片中生成示例文字，并输出文本集合。",
  "modelLabel": "无需模型",
  "icon": "scan-text",
  "modelConfigured": true,
  "launch": {
    "launchType": "python",
    "command": "main.py",
    "args": ["--input", "{inputJson}", "--output", "{outputJson}"],
    "notes": "读取图片集合并输出 OCR 文本。"
  },
  "parameters": []
}
```

核心处理逻辑：

```python
images = input_data.get("inputs", {}).get("images", {}).get("items", [])

results = []
for image in images:
    name = image.get("name", "")
    results.append({
        "path": image.get("path", ""),
        "name": name,
        "relativePath": image.get("relativePath", ""),
        "fullText": f"这里是 {name} 的 OCR 识别文本"
    })

output = {
    "status": "completed",
    "verdict": "pass",
    "message": f"识别 {len(results)} 张图片。",
    "processedFiles": len(images),
    "matchedFiles": len(results),
    "results": results,
    "reportSection": f"### OCR 识别\n\n已识别 {len(results)} 张图片。"
}
```

推荐连接：

```text
待测项目中所有图片 --图片数据--> 示例 OCR
示例 OCR --文本数据--> 文本审核模块
开始 -> 示例 OCR -> 文本审核模块 -> 输出结果
```

关键点：

- OCR 模块类型必须是 `image_ocr`。
- 输出里要写 `results[].fullText`。
- 后续文本模块才能通过文本数据线收到 OCR 结果。

## 完整 main.py 模板

下面模板可以作为所有 Python 模块的起点：

```python
import argparse
import json
from pathlib import Path


def load_json(path):
    return json.loads(Path(path).read_text(encoding="utf-8"))


def write_json(path, value):
    Path(path).write_text(json.dumps(value, ensure_ascii=False, indent=2), encoding="utf-8")


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    parser.add_argument("--output", required=True)
    args = parser.parse_args()

    input_data = load_json(args.input)

    output = {
        "status": "completed",
        "verdict": "pass",
        "message": "模块运行完成。",
        "processedFiles": 0,
        "matchedFiles": 0,
        "artifactCount": 0,
        "reportSection": "### 模块结果\n\n模块运行完成。"
    }

    write_json(args.output, output)


if __name__ == "__main__":
    main()
```
